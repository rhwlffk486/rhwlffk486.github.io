hash.put("user_ID" + i, userIDarr.get(i));
}
 
if (userIDarr.isEmpty()) {
 
redirect.addFlashAttribute("msg", 2);
return "redirect:/user/findUserInfo";
 
} else {
 
sendEmail(userIDarr, hash, "findID");
redirect.addFlashAttribute("msg", 1);
return "redirect:/";
 
}
}
